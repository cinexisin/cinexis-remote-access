#include <tunables/global>

  /share/cinexis/**                 rw,

  /var/lib/nginx/                   rw,

  /run/nginx/                       rw,

  /lib/**                           r,

  /usr/share/ca-certificates/       r,
  /usr/share/ca-certificates/**     r,

  # s6 / init
  /run/s6/                          rw,
  /run/s6/**                        rw,
  /var/run/                         rw,
  /var/run/**                       rw,
}
